let n = 3;
let m = 4;
let count=0;
var arr1 = Array.from({
    length: n // Create 5 indexes with undefined values
  },
  function(v, k) { // Run a map function on said indexes using v(alue)[undefined] and k(ey)[0 to 4]
    return k+1; // Return k(ey) as value for this index
  }
);


var arr2 = Array.from({
    length: m // Create 5 indexes with undefined values
  },
  function(v, k) { // Run a map function on said indexes using v(alue)[undefined] and k(ey)[0 to 4]
    return k+1; // Return k(ey) as value for this index
  }
);



for(i=0;i<n;i++){
  for(j=0;j<m;j++){
  
    var sum = parseInt(arr1[i])+parseInt(arr2[j]);
    
    if(sum % 5 == 0){
      count++;
    }
  }
}

return count;
